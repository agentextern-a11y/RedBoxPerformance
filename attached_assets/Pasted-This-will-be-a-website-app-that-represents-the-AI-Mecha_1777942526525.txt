from fastapi import FastAPI, WebSocket, UploadFile

app = FastAPI()

recorder = InfluxRecorder()
twin = DigitalTwin()
root_ai = RootCauseAI()
predictor = FailurePredictor()
map_vault = MapVault("map_vault.db")
# --- Live Data WebSocket (car + phone) ---
@app.websocket("/ws/live")
async def live_data(ws: WebSocket):
    await ws.accept()
    while True:
        # Merge OBD and phone sensor data
        car_data = await obd_mgr.get_live_data()
        phone_data = await sensor_fusion.get_fused_state()
        merged = {**car_data, **phone_data}
        await recorder.write(merged)  # to InfluxDB
        await ws.send_json(merged)
        await asyncio.sleep(0.01)  # ~100 Hz
# --- Digital Twin Anomaly Check ---
@app.get("/api/health/check")
async def health_check():
    recent_data = recorder.query_last_hours(1)
    anomaly_score = twin.detect_anomaly(recent_data)
    return {"score": anomaly_score, "alert": anomaly_score > 0.85}
# --- Root Cause Diagnosis ---
@app.post("/api/diagnose")
async def diagnose(dtc: str):
    result = root_ai.diagnose(dtc)
    return result  # JSON with graph nodes
# --- Predictive Timeline ---
@app.get("/api/predict/{component}")
async def predict(component: str):
    km_left = predictor.predict_km(component)
    return {"component": component, "km_remaining": km_left}
# --- Map Vault ---
@app.get("/api/maps/{vin}")
async def get_stock_map(vin: str):
    map_data = map_vault.get_map(vin)
    return map_data

async def upload_map(file: UploadFile, vin: str, cal_id: str):

🔌 3. OBD Interface & Sensor Fusion
backend/obd_interface.py



import obd
import asyncio
class OBDManager:
    def __init__(self):
        self.conn = obd.Async(fast=True, timeout=30)
        self.pids = [
            obd.commands.RPM, obd.commands.SPEED, obd.commands.COOLANT_TEMP,
            obd.commands.INTAKE_TEMP, obd.commands.TIMING_ADVANCE,
            obd.commands.FUEL_STATUS, obd.commands.O2_B1S1
        ]
    async def get_live_data(self):
        responses = {}
        for pid in self.pids:
            val = await self.conn.query(pid)
            responses[pid.name] = val.value.magnitude if not val.is_null() else None
        return responses
backend/sensor_fusion.py



class PhoneSensorFusion:
    def __init__(self):
        self.alpha = 0.98  # complementary filter
        self.pitch = 0.0
        self.roll = 0.0
    async def update_phone_data(self, imu, gps):
        # simple slope from accelerometer + gyro
        dt = imu['dt']
        pitch_acc = math.atan2(imu['accel']['z'], 
                  math.sqrt(imu['accel']['x']**2 + imu['accel']['y']**2))
        pitch_gyro = imu['gyro']['beta'] * dt
        self.pitch = self.alpha * (self.pitch + pitch_gyro) + (1-self.alpha)*pitch_acc
        return {"slope_deg": math.degrees(self.pitch), "gps_speed": gps['speed']}

import xgboost as xgb
class FailurePredictor:
    def predict_km(self, component):
        # pre-trained XGBoost model with survival analysis
        features = self.get_recent_features(component)
        dtest = xgb.DMatrix(features)
        km_left = self.model.predict(dtest)
        return int(km_left[0])

frontend/src/MapViewer.jsx – stock map hex viewer + download



import { useState } from 'react';
export default function MapViewer({ vin }) {
  const [mapData, setMapData] = useState(null);
  const loadMap = async () => {
    const res = await fetch(`/api/maps/${vin}`);
    const data = await res.json();
    setMapData(data);
  };
  const download = () => {
    const blob = new Blob([mapData.binary], { type: 'application/octet-stream' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url; a.download = `${vin}_stock.bin`; a.click();
  };
  return (
    <div>
      <button onClick={loadMap}>Load Stock Map</button>
      {mapData && (
        <>
          <HexViewer raw={mapData.raw} />
          <button onClick={download}>Download .bin</button>
          <Map3DVisualizer data={mapData.tables} />
        </>
      )}
    </div>
  );
}
🗄 6. Map Vault Logic
backend/map_vault.py



import sqlite3, os
class MapVault:
    def __init__(self, db_path):
        self.conn = sqlite3.connect(db_path)
        self.conn.execute('''CREATE TABLE IF NOT EXISTS stock_maps
            (vin TEXT, cal_id TEXT, file_path TEXT, checksum TEXT)''')
    def get_map(self, vin):
        row = self.conn.execute("SELECT * FROM stock_maps WHERE vin=?", (vin,)).fetchone()
        if not row: return None
        with open(row[2], 'rb') as f:
            raw = f.read()
        return {"vin": vin, "cal_id": row[1], "binary": raw.hex(), "checksum": row[3]}
