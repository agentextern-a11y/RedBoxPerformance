· The frontend handles all OBD communication natively via navigator.bluetooth, so the user just clicks “Connect” – the browser does the rest.
· Phone sensor data (GPS, accelerometer) is acquired by opening a companion page on the smartphone (served by the same backend) that streams to the backend over WebSocket.
· The backend ingests all streams, records data, runs AI inference, and stores the map vault.
· Everything lives in Docker containers, ready for local deployment, but also portable to a home server or Raspberry Pi.

import numpy as np
import tensorflow as tf
class DigitalTwin:
    def __init__(self):
        self.model = tf.keras.models.load_model('models/lstm_autoencoder.h5')
        self.buffer = []
        self.threshold = 0.12
    def ingest(self, live_data):
        sample = np.array([[
            live_data.get('rpm', 0),
            live_data.get('speed', 0),
            live_data.get('coolant_temp', 0),
            live_data.get('intake_temp', 0),
            live_data.get('timing', 0),
            live_data.get('o2_voltage', 0),
            live_data.get('throttle', 0)
        ]])
        reconstruction = self.model.predict(sample, verbose=0)
        mse = np.mean(np.square(sample - reconstruction))
        return float(mse)  # 0 = perfect, high = anomaly

Key Magic: useBluetooth.ts – Web Bluetooth OBD Connection (No Backend Needed)



// frontend/src/hooks/useBluetooth.ts
import { useState, useCallback } from 'react';
const OBD_SERVICE_UUID = '0000fff0-0000-1000-8000-00805f9b34fb'; // ELM327 common
const OBD_CHAR_UUID = '0000fff1-0000-1000-8000-00805f9b34fb';
export function useBluetooth() {
  const [device, setDevice] = useState<BluetoothDevice | null>(null);
  const [server, setServer] = useState<BluetoothRemoteGATTServer | null>(null);
  const [char, setChar] = useState<BluetoothRemoteGATTCharacteristic | null>(null);
  const [connected, setConnected] = useState(false);
  const connect = useCallback(async () => {
    try {
      const dev = await navigator.bluetooth.requestDevice({
        filters: [{ namePrefix: 'VLINK' }],  // or 'OBDII'
        optionalServices: [OBD_SERVICE_UUID]
      });
      const srv = await dev.gatt!.connect();
      const service = await srv.getPrimaryService(OBD_SERVICE_UUID);
      const characteristic = await service.getCharacteristic(OBD_CHAR_UUID);
      setDevice(dev);
      setServer(srv);
      setChar(characteristic);
      setConnected(true);
    } catch (err) {
      console.error('BLE connection failed:', err);
      // If Web Bluetooth not supported, show fallback message
    }
  }, []);
  const sendCommand = useCallback(async (cmd: string) => {
    if (!char) return;
    const encoder = new TextEncoder();
    await char.writeValue(encoder.encode(cmd + '\r'));
  }, [char]);
  const readResponse = useCallback(async (): Promise<string> => {
    if (!char) return '';
    const value = await char.readValue();
    return new TextDecoder().decode(value);
  }, [char]);
  return { connected, connect, sendCommand, readResponse, device };
}

frontend/src/components/MapViewer.tsx



import { useState } from 'react';
export default function MapViewer() {
  const [vin, setVin] = useState('');
  const [map, setMap] = useState<any>(null);
  const loadMap = async () => {
    const res = await fetch(`http://localhost:8000/api/maps/${vin}`);
    const data = await res.json();
    setMap(data);
  };
  const downloadBin = () => {
    if (!map?.raw) return;
    const blob = new Blob([new Uint8Array(map.raw.match(/.{2}/g).map((b: string) => parseInt(b, 16)))]);
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `${vin}_stock.bin`;
    a.click();
    URL.revokeObjectURL(url);
  };
  return (
    <div className="map-viewer">
      <input value={vin} onChange={e => setVin(e.target.value)} placeholder="VIN" />
      <button onClick={loadMap}>Load Stock Map</button>
      {map && (
        <>
          <div className="hex-view">{map.raw}</div>
          <button onClick={downloadBin}>Download .bin</button>
        </>
      )}
    </div>
  );
}
Backend map vault logic (backend/map_vault.py):



import sqlite3, os, hashlib
from fastapi import UploadFile
class MapVault:
    def __init__(self, db_path="data/maps/map_vault.db"):
        self.db_path = db_path
        self.conn = sqlite3.connect(db_path)
        self.conn.execute('''CREATE TABLE IF NOT EXISTS maps
            (vin TEXT PRIMARY KEY, cal_id TEXT, file_path TEXT, checksum TEXT)''')
        self.conn.commit()
    def get_map(self, vin: str):
        row = self.conn.execute("SELECT * FROM maps WHERE vin=?", (vin,)).fetchone()
        if not row:
            return {"error": "Map not found"}
        with open(row[2], "rb") as f:
            raw = f.read()
        return {"vin": vin, "cal_id": row[1], "raw": raw.hex(), "checksum": row[3]}
    async def store(self, vin: str, cal_id: str, file: UploadFile):
        checksum = hashlib.md5(await file.read()).hexdigest()
        file.file.seek(0)
        os.makedirs("data/maps", exist_ok=True)
        file_path = f"data/maps/{vin}.bin"
        with open(file_path, "wb") as f:
            f.write(await file.read())
        self.conn.execute("INSERT OR REPLACE INTO maps VALUES (?,?,?,?)",
                          (vin, cal_id, file_path, checksum))
        self.conn.commit()
