packages/sdk/src/index.ts

const API = "http://localhost:3001/metrics"
export const startTracking = () => {
  const send = async (metric: any) => {
    try {
      await fetch(API, {
        method: "POST",
        headers: {"Content-Type":"application/json"},
        body: JSON.stringify(metric)
      })
    } catch {}
  }
  if ("PerformanceObserver" in window) {
    const obs = new PerformanceObserver((list) => {
      list.getEntries().forEach(e => {
        if (e.duration > 50) {
          send({ type:"longtask", value:e.duration })
        }
      })
    })
    obs.observe({ entryTypes:["longtask"] })
  }
  window.onerror = () => send({ type:"error", value:1 })
}

let history:number[] = []
export const detect = (v:number) => {
  history.push(v)
  if (history.length < 5) return null
  const avg = history.reduce((a,b)=>a+b,0)/history.length
  if (v > avg * 1.7) {
    return { level:"high", message:"Performance anomaly" }
  }
}

  if (m.type==="longtask") {
    return m.value > 120
      ? "Heavy blocking detected"
      : "Optimize main thread"
  }
  if (m.type==="error") return "Runtime error detected"
}

apps/web/src/App.tsx

import { useEffect, useState } from "react"
import { startTracking } from "sdk"
export default function App(){
  const [logs,setLogs]=useState<any[]>([])
  useEffect(()=>{
    startTracking()
    const i=setInterval(async()=>{
      const res=await fetch("http://localhost:3001/sessions")
      const data=await res.json()
      setLogs(data)
    },1000)
    return ()=>clearInterval(i)
  },[])
  return(
    <div style={{background:"#0b0f17",color:"#fff",padding:20}}>
      <h1>🔴 RedBoxX Pro</h1>
      {logs.map((l,i)=>(
        <div key={i}>
          {l.type} {l.value}ms
          {l.analysis && <div style={{color:"red"}}>{l.analysis.issue}</div>}
        </div>
      ))}
    </div>
  )
}

server/index.ts

import Fastify from "fastify"
import cors from "@fastify/cors"
const app = Fastify()
await app.register(cors,{origin:true})
const db:any[] = []
function analyze(m:any){
  if(m.value>120){
    return {
      severity:"high",
      issue:"Main thread blocking",
      fix:"Split heavy tasks"
    }
  }
}
app.post("/metrics", async(req,res)=>{
  const body:any=req.body
  const analysis=analyze(body)
  const entry={...body,analysis,time:Date.now()}
  db.push(entry)
  return {ok:true}
})
app.get("/sessions", async()=>{
  return db.slice(-50)
})
app.listen({port:3001},()=>{
  console.log("🚀 API running")
})
